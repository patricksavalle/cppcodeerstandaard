Regel: enum-test

Elk switch-statement dat test op een enumerated-type moet voor elke waarde van de enumerated een case-label bevatten.

Houd deze regel wel praktisch; een enumerated-type met vele tientallen verschillende waarden hoeft natuurlijk niet aan deze regel te voldoen. Een voorbeeld: 

  enum WeekDay
  { NULL_DAY=-1
  , DAY_SUNDAY=0
  , DAY_MONDAY
  , DAY_TUESDAY
  , DAY_WEDNESDAY
  , DAY_THURSDAY
  , DAY_FRIDAY
  , DAY_SATERDAY 
  };

  bool IsWorkingDay( WeekDay day )
  {
    bool ret = false;

    switch(day)
    {
      case DAY_SUNDAY:
      case DAY_SATERDAY:
        ret = false;
        break;

      case DAG_MONDAY:
      case DAG_TUESDAY:
      case DAG_WEDNESDAY:
      case DAG_THURSDAY:
      case DAG_FRIDAY:
        ret = true;
        break;

      case NULL_DAY:
      default:
        throw XSomeException( "invalid value" );
        break;       
    }

    return ret;
  }
